    <script>
      function setJSON(key, data) {
        if (typeof data != "string") {
          data = JSON.stringify(data);
        }
        localStorage.setItem(key, data);
      }

      function getJSON(key) {
        return JSON.parse(localStorage.getItem(key));
      }
    </script>
